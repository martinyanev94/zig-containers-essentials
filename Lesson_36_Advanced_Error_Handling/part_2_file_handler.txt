const std = @import("std");

pub const FileError = error{
    NotFound,
    PermissionDenied,
    ReadError,
};

fn openFile(filename: []const u8) !*std.fs.File {
    var file = std.fs.File.openRead(filename) catch |err| {
        switch (err) {
            error.FileNotFound => return FileError.NotFound,
            error.PermissionDenied => return FileError.PermissionDenied,
            else => return error.UnknownError,
        }
    };
    return file;
}

pub fn main() !void {
    const file = try openFile("example.txt");
    defer file.close();
    std.debug.print("File opened successfully: {}\n", .{file.filename});
}
