const std = @import("std");

pub fn square(value: u32) u32 {
    var result: u32 = 0;

    asm (
        "imul %1, %0" // Multiply value by itself
        : "=r"(result) // Output operand
        : "r"(value) // Input operand
    );

    return result;
}

pub fn main() void {
    const number: u32 = 5;
    const squared_value: u32 = square(number);

    std.debug.print("Square of {} is: {}\n", .{number, squared_value});
}
